Q1. You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string. 
        Input:
        abcadeecfb
        
        Output:
        abcdef

let str="abcadeecfb";
        let dup=new Set(str);
        console.log([...dup].sort().join(''))

2.You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps) 
      Input: abcadeecfb 
      Output: a=2 b=2 c=2 d=1 e=2f=1"

let str="abedefsgd"
    let s=new Set(str)
    for(i of s){
        let count=0;
        for(j=0;j < str.length;j++){
            if(i==str[j]){
                count++
            }
        }console.log(i,'->',count)
    }
