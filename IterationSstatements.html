const Find_Five = (N) => 
      {
        let count=0;
        while(N!=0)
        {
          if(N%10==5)
          {
            count++;
          }
          N=parseInt(N/10);
        }

3.Find Sum

const findSum = (n) => {
      let sum=0
      for(let i=2;i<=n;i=i+2)
      {
        sum=sum+i;
      }
      return sum;
  };

const Number_Sum = (N) => 
      {
