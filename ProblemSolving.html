      var Minimal_Angle = (h, m) => 
      {
        const hAngle=(h*30)+(m/2);
        const mAngle=m*6;
        let a=0;
        if(hAngle>mAngle)
        {
           a=hAngle-mAngle;
        }
        else
        {
          a=mAngle-hAngle;
        }
        
        if(a>180)
        {
          a=360-a;
        }
        
          return a;
      };

3.Perfect Number Check.

var Perfect_Check = (N) => 
      {
           let count=1; 
           for(i=2;i<=N/2;i++)
           {
             if(N%i===0)
             {
               count=count+i;
             }
           }
           if(count==N)
           {
             return "YES";
           }
           return "NO";
      };

4.Reverse a Number.

var Reverse_Number = (N) => 
      {
      let x = N.toString();
      let r = x.split("").reverse().join("");
      
      return parseInt(r);
      
      
      };
